```javascript
function getClosure() {
    var canYouSeeMe = "here I am";
    return (function theClosure() {
        return {canYouSeeIt: canYouSeeMe ? "yes!" : "no"};
    });
}

var closure = getClosure();
console.log(closure().canYouSeeIt);  // ?
```
